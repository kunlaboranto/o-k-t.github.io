void handler(int sig)
{
    fprintf(stdout, "signal occured (sig=%d)\n", sig); fflush(stdout);
    // DO IT..
}

void mySignal(void)
{
    signal( SIGHUP, handler);    /* 1 : hangup */
    signal( SIGINT, handler);    /* 2 : interrupt (rubout) */
    signal( SIGQUIT, handler);   /* 3 : quit (ASCII FS) */
    signal( SIGILL, handler);    /* 4 : illegal instruction(not reset when caught) */
    signal( SIGTRAP, handler);   /* 5 : trace trap (not reset when caught) */
    signal( SIGIOT, handler);    /* 6 : IOT instruction */
    signal( SIGABRT, handler);   /* 6 : used by abort,replace SIGIOT in the future */
    signal( SIGEMT, handler);    /* 7 : EMT instruction */
    signal( SIGFPE, handler);    /* 8 : floating point exception */
    signal( SIGKILL , handler);  /* 9 : kill (cannot be caught or ignored) */
    signal( SIGBUS , handler);   /* 10: bus error */
    signal( SIGSEGV, handler);   /* 11: segmentation violation */
    signal( SIGSYS , handler);   /* 12: bad argument to system call */
    signal( SIGPIPE, handler);   /* 13: write on a pipe with no one to read it */
    signal( SIGALRM, handler);   /* 14: alarm clock */
    signal( SIGTERM, handler);   /* 15: software termination signal from kill */
    signal( SIGUSR1, handler);   /* 16: user defined signal 1 */
    signal( SIGUSR2, handler);   /* 17: user defined signal 2 */
    sigignore( SIGCLD );              /* 18: child status change */
    sigignore( SIGCHLD );             /* 18: child status change alias (POSIX) */
    signal( SIGPWR , handler);   /* 19: power-fail restart */
    sigignore( SIGWINCH );            /* 20: window size change */
    signal( SIGURG  , handler);  /* 21: urgent socket condition */
    signal( SIGPOLL , handler);  /* 22: pollable event occured */
    signal( SIGIO   , handler);  /* 22: socket I/O possible (SIGPOLL alias) */
    signal( SIGSTOP , handler);  /* 23: stop (cannot be caught or ignored) */
    signal( SIGTSTP , handler);  /* 24: user stop requested from tty */
    sigignore( SIGCONT );               /* 25: stopped process has been continued */
    signal( SIGTTIN , handler);  /* 26: background tty read attempted */
    signal( SIGTTOU , handler);  /* 27: background tty write attempted */
    signal( SIGVTALRM , handler);/* 28: virtual timer expired */
    signal( SIGPROF , handler);  /* 29: profiling timer expired */
    signal( SIGXCPU , handler);  /* 30: exceeded cpu limit */
    signal( SIGXFSZ , handler);  /* 31: exceeded file size limit */
}
