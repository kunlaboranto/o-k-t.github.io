제가 쓰고 있는 설정 file입니다.

/etc/rc.d/init.d 밑에 두고 ntsysv로 설정해서 쓰세요.



#!/bin/sh
#
# Startup script for oracle
#

# Source function library.
. /etc/rc.d/init.d/functions

# See how we were called.
case "$1" in
start)
echo "Starting oraHFC"
echo "==============="
su - oracle -c "svrmgrl << __EOF__
connect internal
startup
__EOF__"

echo ""
echo "Starting TNS listener"
echo "====================="
su - oracle -c "lsnrctl start"
touch /var/lock/subsys/oraHFC
;;

stop)
echo ""
echo "Shutting down TNS listener"
echo "=========================="
su - oracle -c "lsnrctl stop"

echo ""
echo "Shutting down oraHFC"
echo "===================="
su - oracle -c "svrmgrl << __EOF__
connect internal
shutdown immediate
__EOF__"
rm -f /var/lock/subsys/oraHFC
;;

restart)
$0 stop
$0 start
;;

*)
echo "Usage: $0 {start|stop|restart}"
exit 1
esac
exit 0
