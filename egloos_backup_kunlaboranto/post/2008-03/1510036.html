altibase 추천 ~/.aliases 예

## altibase 추천 ~/.aliases 예

$ cat ~/.aliases

export _HOSTNAME=`hostname`
PS1='
$_HOSTNAME: $PWD
$ '

alias h='history | tail -20'

alias ls='ls -F'
alias ll='ls -l'
alias lh='ls -alt | head'

alias pe='ps -ef | grep -i -e PID -e '

#alias vi='vim'
alias hdiff='vim -d -o2'

alias ath='cd $ALTIBASE_HOME'
alias atb='cd $ALTIBASE_HOME/bin'
alias ats='cd $ALTIBASE_HOME/sample'
alias atp='vi $ALTIBASE_HOME/conf/altibase.properties'
alias att='tail -f $ALTIBASE_HOME/trc/altibase_boot.log'    
alias attsm='tail -f $ALTIBASE_HOME/trc/altibase_sm.log' 
alias attrp='tail -f $ALTIBASE_HOME/trc/altibase_rp.log' 
alias atl='find /ALTILOG/logs/. -name "logfile*" -print | wc -l'
