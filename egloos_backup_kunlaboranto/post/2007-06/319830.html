analytic fuction - row_number() over ..

-- oracle 9i 이후 제공되는 analytic 함수는 현재 지원되지 못함.

SELECT ANALYZER_ID,INTERFACE,port, rank
FROM (
 SELECT ROW_NUMBER() OVER (
  PARTITION BY ANALYZER_ID,INTERFACE
  ORDER BY ANALYZER_ID,INTERFACE,CNT DESC,TOTALBYTES DESC
 ) RANK ,A.*
 FROM P_MONTH_tcpsrcport_TB A
 WHERE DB_COLLECT_TM = '200609'
) INV
WHERE RANK < 6
ORDER BY ANALYZER_ID,INTERFACE,CNT DESC,TOTALBYTES DESC;
