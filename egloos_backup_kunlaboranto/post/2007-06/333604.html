[script] 멋진 awk - 범위 조각 모음

작업중, 다른 분이 쓰는 것을 보고, 신기해 하면서 적어둠
범위 BEGIN ~ END 값들이 있을때, 연결가능한 범위를 합치는 스크립트

--- 실행예
$ cat unsort.txt
1010 1020
1001 1010
1010 1010
3001 3010
2010 2020
2001 2010
2010 2010

$ imei.sh unsort.txt

1001  1020
2001  2020
3001  3010

           7 row ,          54 imei ,           4 list

$ cat imei.sh
cat $1 |sort |grep -v row |grep [0-9] |awk '{
ne=en+1
cnt=cnt+1
c=$2-$1+1
cc=cc+c
if($1 == en || $1 == ne){
en=$2
}
else {
print en
printf("%s  ",$1)
en=$2
ccc=ccc+1
}
}
END{
print en
ccc=ccc+1
printf("\n\n  %10d row ,  %10d imei ,  %10d list \n",cnt,cc,ccc)
}'
