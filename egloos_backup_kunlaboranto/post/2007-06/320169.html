윈도에서 DB관련 CLASS 예

윈도 프로그래밍에 맹 인 쥔장으로서는..

여러 다양한 DB관련 윈도 컴포넌트 (CLASS)들이 혼동 스럽기 짝이 없음.

 

때문에.. 명칭 및 접속 사용예 + 오류처리예 정도로 수집시작함.

 

#1. C#에서 CDatabase

 CDatabase database;
 CString sDsn;

 CString SqlString; 
 
 SqlString.Format("INSERT INTO test3 VALUES ('MFC', '%d')", thr_no);
 sDsn.Format(_T("ODBC;DSN=A4;UID=SYS;PASSWORD=MANAGER"));

 

 TRY
 {
  database.Open(NULL, false, false, sDsn);
  while(1)
  {
   Sleep(100);
   database.ExecuteSQL(SqlString);
  }

 }
 CATCH(CDBException, e)
 {
  AfxMessageBox("Database error: "+e->m_strError);
  //AfxMessageBox("_OK_01 : " + e->m_nRetCode);
  AfxMessageBox("_OK_01 : " + e->m_strStateNativeOrigin);

 }
