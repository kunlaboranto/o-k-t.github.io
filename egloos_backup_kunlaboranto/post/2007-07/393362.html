Kermit Howto

# Hongki's .kermrc from HLFTO & HONGKI
# 이 환경설정 파일은 하이텔의 이선중(HLFTO)님이 작성하신 것입니다.
cd  \v(home)
#set line /dev/modem
#set speed 2400
#set modem hayes
set rec pack 9024
set send pack  9024
set window 5
set term byte 8
set com byte 8
set telnet echo remote
set TELNET terminal-type vt100
set TELNET newline-mode on 
#set host chollian.dacom.co.kr
#set host hitel.goldstar.co.kr
set terminal character-set cp437 cp437
set file bytesize 8
set file character-set cp437
set file type bin
set file name lit
set file display full
set file collision update
set windows-size 1
set terminal byte 8
set terminal char transparent
set parity none
set block-check 3
set rec pack 2000
set send pack 2000
set window 5
set retry-limit 47
set prompt Hongki Kermit> 
set key \127 \008
COMMENT - Edit (vi)
;
def vi if = \v(argc) 2 assign myfile \%1,-
  if not def myfile echo Edit what?,-
  else run vi \m(myfile),-
  def myfile
COMMENT - Capture (cap)
;
def cap log session captured append, output \13, connect
def uncap close session, connect
COMMENT - View (less, more)
;
def less if = \v(argc) 2 assign myfile \%1,-
  if not def myfile echo View what?,-
  else run less \m(myfile),-
  def myfile
def more less \%1
 
COMMENT - RzSz
;
def rz run /usr/bin/rz </dev/modem>/dev/modem, output \13, connect
def sz if = \v(argc) 1 end 1 {Send what file(s)?}, -
    run /usr/bin/sz +b \%1 \%2 \%3 \%4 \%5 \%6 \%7 \%8 \%9 -
    </dev/modem>/dev/modem
COMMENT - Dial Directory
;
if exist \v(home).kermdd -
  set dial directory \v(home).kermdd ; C-Kermit dial directory
COMMENT - Greeting.
;
run clear
if < \v(ntime) 43200 echo Good Morning!
  else if < \v(ntime) 61200 echo Good Afternoon!
  else echo Good Evening.
echo 
; End of C-Kermit 5A initialization file.

*. Kermit 명령어 정리
        Kermit Prompt상에서
                c, connect ; c를 누르면 접속화면으로 바뀐다.
                set ; 환경 설정을 바꿀 수 있다. 
                        ex) KERMIT> set host nowcom.co.kr
                            KERMIT> c
                            (나우누리로 접속한다.)
        
                quit ; kermit을 종료한다.
                ! ; shell로 잠시 나간다. (exit로 돌아온다.)
        
        접속화면에서
                ctrl-\ c ; [CTRL]키와 [\]키를 동시에 눌러주고 [c]를 누른다.
                           다시 Kermit Prompt로 나오게 된다.
                           돌아갈때는 c를 누르고 엔터친다.
