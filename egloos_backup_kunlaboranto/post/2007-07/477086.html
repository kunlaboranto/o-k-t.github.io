SELECT
  base_time - decode(idle_start_time, 0, base_time, idle_start_time) idle_time,
  CASE2( IDLE_START_TIME = 0, '0',
    TO_CHAR(TO_DATE('1970010109','YYYYMMDDHH') + IDLE_START_TIME/60/60/24 ,
    'MM/DD HH:MI:SS') )
  ||', '||ID as 'idle_time (ss_id)',
  substr(comm_name, 22, 20) || ' ('|| client_pid || ')' as 'IP (PID)'
FROM V$SESSION , v$sessionmgr
order by IDLE_START_TIME
limit 30
;
